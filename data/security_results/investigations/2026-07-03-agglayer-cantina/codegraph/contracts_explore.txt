232	/**
233	 * Derive gas token address from rollup ID by repeating it 5 times to get 160 bits
234	 */
235	function deriveGasTokenAddress(rollupID: number): string {
236	    // Convert rollupID to 32-bit hex (8 characters)
237	    const rollupHex = rollupID.toString(16).padStart(8, '0').toLowerCase();
238	
239	    // Repeat 5 times to get 160 bits (40 hex characters = 20 bytes)
240	    const addressHex = rollupHex.repeat(5);
241	    const address = `0x${addressHex}`;
242	
243	    return address;
244	}
